num proc factorial(num x)
[
	if(x<0)
	{
		out 0;
	};
	if(x&0)
	{
		out 1;
	}
	else
	{
		create num temp;
		temp = x - 1;
		out factorial(temp)*x;
	};
];

num proc sunlight(word a, word b)
[
	create num q;
	q = lexStrCmp(a,b);
	out q;
];
